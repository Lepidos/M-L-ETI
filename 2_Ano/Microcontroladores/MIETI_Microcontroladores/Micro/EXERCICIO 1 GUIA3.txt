#define NLOOPS 20

	
	LJMP MAIN

	CSEG AT 30H
MAIN:
	MOV R1,80H//LOCAL ONDE OS NROS PARES VAO FICAR
	MOV R2,0A0H//LOCAL ONDE OS NUMEROS IMPARES VAO FICAR
	
	MOV R0,#30H //COLOCAR O NUMERO QUE ESTA NA POSIÇÃO 30H NO REGISTO 0
	MOV R3,#NLOOPS //COLOCAR NO REGISTO 1 O CONTEUDO DE NLOOPS
	
LOOP:
	MOV A,@R0// O CONTEUDO DE R0 É CARREGADO PARA O ACUMULADOR
	INC R0// INCREMENTA A POSICAO 30H 
	DJNZ R3,VERIFPAR
	JMP FINISH
		
		
VERIFPAR:	
		CLR C// LIMPAR O CARRY
		RRC A//FAZ UM SHIFT RIGHT A TODOS OS BITS E O DA DIREITA FICA NO CARRY
		JC VERIFIMP// SE O VALOR DO CARRY ESTIVER A 1, ENTAO É IMPAR E SALTA PARA VERIFIMP
		MOV @R1,A// VISTO QUE O VALOR E PAR ENTAO VAI PARA O ENDEREÇO DE MEMORIA ONDE DEVE FICAR OU SEJA O 80H
		INC R2//INCREMENTAR A POSICAO DE MEMORIA PARA NAO HAVER SOBREPOSIÇÃO DE DADOS
		JMP LOOP
		
		
VERIFIMP:
        MOV R1,0A0H
		MOV @R1,A// O NUMERO É IMPAR, MANDO-O PARA O LUGAR DOS IMPARES
		INC R1// INCREMENTO A POSIÇAO DE MEMORIA DOS IMPARES PARA NAO HAVER SOBREPOSICAO
		JMP LOOP
		
FINISH:
		SJMP $
		END
