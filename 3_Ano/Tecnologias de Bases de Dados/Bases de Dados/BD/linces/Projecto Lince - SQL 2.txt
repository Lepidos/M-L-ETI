SELECT n_func, funcionario from TECNICOS
	WHERE n_func IN (Select C.n_func from CONTROLOS C, LINCES L
			WHere C.id_lince=L.id_lince
			AND L.genero="M"
			AND C.data BETWEEN "2014-02-01" AND "2014-02-28")
	AND n_func NOT IN (SELECT C.n_func FROM CONTROLOS C,LINCES L
			WHERE C.id_lince=L.id_lince 
			AND L.genero="F"
			AND C.data BETWEEN "2014-02-01" AND "2014-02-28");
